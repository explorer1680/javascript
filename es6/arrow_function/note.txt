arrow function get this when it created.
when defined in object:
    {
        a: 'a'
        f: ()=>{
            this means windows
        }
    }

when defined in class:
    class C {
        m = () =>{
            this means this instance of C
        }
    }

when defined in function:
    var P = function(...){
        this.f = ()=>{
            this means the object.
        }
    }

when defined in prototype:
    p.prototype.f=() =>{
        this means windows
    }
